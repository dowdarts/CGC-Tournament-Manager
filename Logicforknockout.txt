1. Calculating the Bracket Size
You first need to find the "Next Power of 2."

Dart
import 'dart:math';

int getBracketSize(int playerCount) {
  if (playerCount <= 2) return 2;
  // find the exponent: log2 of playerCount, rounded up
  return pow(2, (log(playerCount) / log(2)).ceil()).toInt();
}
2. Seeding with "Byes"
In a standard tournament, "Byes" are assigned to the highest seeds so they get the advantage of playing one fewer match.

Here is the updated logic for your KnockoutBracket class:

Dart
void generateFirstRound() {
  int bracketSize = getBracketSize(topPlayers.length);
  int numByes = bracketSize - topPlayers.length;

  // 1. Create a list of players, padding the end with 'null' for byes
  List<Player?> seededSlots = List<Player?>.from(topPlayers);
  for (int i = 0; i < numByes; i++) {
    seededSlots.add(null); 
  }

  // 2. Pair High-Low (1 vs 8, 2 vs 7, etc.)
  currentRoundMatches = [];
  for (int i = 0; i < bracketSize / 2; i++) {
    Player? p1 = seededSlots[i];
    Player? p2 = seededSlots[bracketSize - 1 - i];

    Match match = Match(player1: p1, player2: p2);
    
    // 3. Logic: If one side is null, the other player is an automatic winner
    if (p1 == null || p2 == null) {
      match.winner = p1 ?? p2;
      match.isBye = true; // Helpful flag for your UI
    }
    
    currentRoundMatches.add(match);
  }
  allRounds.add(currentRoundMatches);
}

2. The Multi-Round Logic
Instead of generating one round at a time, it is often better to pre-calculate the entire tree. This way, the user can see the whole bracket immediately.

Dart
class TournamentController {
  List<List<Match>> rounds = [];

  void initializeBracket(List<Player> players) {
    int size = getBracketSize(players.length); // Power of 2 logic from before
    int totalRounds = (log(size) / log(2)).floor();
    
    // 1. Generate empty matches for all rounds
    for (int r = 0; r < totalRounds; r++) {
      int matchesInRound = size ~/ pow(2, r + 1);
      rounds.add(List.generate(matchesInRound, (i) => Match(id: "R$r-M$i")));
    }

    // 2. Seed the first round (High-Low)
    _seedFirstRound(players, size);
  }

  void _seedFirstRound(List<Player> players, int size) {
    for (int i = 0; i < size / 2; i++) {
      var p1 = i < players.length ? players[i] : null;
      var p2 = (size - 1 - i) < players.length ? players[size - 1 - i] : null;

      var match = rounds[0][i];
      match.player1 = p1;
      match.player2 = p2;

      if (p1 == null || p2 == null) {
        match.isBye = true;
        match.winner = p1 ?? p2;
        _advanceWinner(0, i, match.winner!);
      }
    }
  }

  void _advanceWinner(int currentRoundIndex, int matchIndex, Player winner) {
    if (currentRoundIndex + 1 >= rounds.length) return; // Champion crowned

    int nextMatchIndex = matchIndex ~/ 2;
    var nextMatch = rounds[currentRoundIndex + 1][nextMatchIndex];

    // If matchIndex is even, they go to player1 slot, if odd, player2 slot
    if (matchIndex % 2 == 0) {
      nextMatch.player1 = winner;
    } else {
      nextMatch.player2 = winner;
    }
  }
}

2. Saving to Windows Local Storage
For Windows, we use the path_provider package to find the "Application Documents" folder. This ensures the tournament data is stored in a standard location.

Dart
import 'dart:io';
import 'dart:convert';
import 'package:path_provider/path_provider.dart';

class TournamentStorage {
  Future<String> get _localPath async {
    final directory = await getApplicationDocumentsDirectory();
    return directory.path;
  }

  Future<File> get _localFile async {
    final path = await _localPath;
    return File('$path/tournament_data.json');
  }

  Future<void> saveTournament(List<List<Match>> rounds) async {
    final file = await _localFile;
    // Map the nested list to a format JSON likes
    String jsonContent = jsonEncode(
      rounds.map((round) => round.map((m) => m.toJson()).toList()).toList()
    );
    await file.writeAsString(jsonContent);
  }

  Future<List<List<Match>>?> loadTournament() async {
    try {
      final file = await _localFile;
      if (!await file.exists()) return null;

      String contents = await file.readAsString();
      List<dynamic> decoded = jsonDecode(contents);

      return decoded.map((round) => 
          (round as List).map((m) => Match.fromJson(m)).toList()
      ).toList();
    } catch (e) {
      return null;
    }
  }
}
