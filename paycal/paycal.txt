            if (rankA &gt; rankB) return 1;
            return 0; // Should not happen with well-defined ranks
        }

        function applyPreset(presetName) {
            const entryFee = parseFloat(document.getElementById(&#39;entryFee&#39;).value) || 0;
            const numPlayers = parseInt(document.getElementById(&#39;numPlayers&#39;).value) || 0;

            if (presetName === &#39;top4_buyin_refund&#39;) {
                // Top 4 is a special case: 50/25/12.5/12.5 split
                prizes = PRESETS[presetName].map(p =&gt; ({ ...p }));
            } else if (presetName.endsWith(&#39;_buyin_refund&#39;)) {
                // Dynamic Money Back logic for Top 8, 16, 32
                const n = parseInt(presetName.match(/\d+/)[0]);
                const ranks = PRESETS.createRanks(n);

                // Need to use the original entry fee for the refund calculation (Refund is based on GROSS entry fee)
                const numRefundPlayers = ranks[ranks.length - 1].split; // e.g., 4 for top 8, 8 for top 16
                const totalRefundAmount = numRefundPlayers * entryFee; 

                // Get the net prize pool for scaling, which respects the current deduction setting
                
                const dartConnectEnabled = document.getElementById(&#39;enableDartConnectFee&#39;).checked;
                const highScoreEnabled = document.getElementById(&#39;enableHighScoreCont&#39;).checked;
                const highFinishEnabled = document.getElementById(&#39;enableHighFinishCont&#39;).checked;

                const dartConnectInput = parseFloat(document.getElementById(&#39;dartConnectFee&#39;).value) || 0;
                const highScoreInput = parseFloat(document.getElementById(&#39;highScoreCont&#39;).value) || 0;
                const highFinishInput = parseFloat(document.getElementById(&#39;highFinishCont&#39;).value) || 0;

                const dartConnectFee = dartConnectEnabled ? dartConnectInput : 0;
                const highScoreCont = highScoreEnabled ? highScoreInput : 0;
                const highFinishCont = highFinishEnabled ? highFinishInput : 0;
                
                const totalDeductionPerPlayer = dartConnectFee + highScoreCont + highFinishCont;
                const netEntryFee = Math.max(0, entryFee - totalDeductionPerPlayer);
                const netMainPrizePool = netEntryFee * numPlayers;


                if (netMainPrizePool === 0) {
                    prizes = []; // Cannot calculate if pool is 0
                    calculatePrizes();
                    return;
                }

                // Calculate the percentage needed for the refund tier (based on NET pool)
                const lastPercent = (totalRefundAmount / netMainPrizePool) * 100;
                
                // Determine the percentage remaining for the top places
                const remainingPercent = 100 - lastPercent;

                // Define standard ratios for the top ranks after reserving the refund amount (30% Winner base)
                const ratios = {
                    &#39;top8&#39;: [30.00, 20.00, 20.00], // Ratios for 1st, 2nd, 3rd-4th
                    &#39;top16&#39;: [30.00, 15.00, 15.00, 20.00], // Ratios for 1st to 5th-8th
                    &#39;top32&#39;: [30.00, 15.00, 10.00, 15.00, 15.00], // Ratios for 1st to 9th-16th
                };
                const ratioKey = presetName.substring(0, presetName.indexOf(&#39;_&#39;));
                const baseRatios = ratios[ratioKey] || [];
                const baseRatioTotal = baseRatios.reduce((sum, ratio) =&gt; sum + ratio, 0);

                prizes = [];

                if (baseRatioTotal &gt; 0 &amp;&amp; remainingPercent &gt; 0) {
                    // 4. Scale up the top rank percentages using the remaining pool
                    for (let i = 0; i &lt; baseRatios.length; i++) {
                        const scaledPercent = (baseRatios[i] / baseRatioTotal) * remainingPercent;
                        prizes.push({ place: ranks[i].place, percent: scaledPercent, split: ranks[i].split });
                    }
                } else if (remainingPercent &gt; 0 &amp;&amp; baseRatioTotal === 0) {
                     // Fallback for cases where baseRatios is empty but there&#39;s a remaining pool
                    prizes.push({ place: &#39;1st&#39;, percent: remainingPercent, split: 1 });
                }
                
                // 5. Add the final refund tier (if lastPercent is positive)
                if (lastPercent &gt; 0) {
                    prizes.push({ place: ranks[ranks.length - 1].place, percent: lastPercent, split: ranks[ranks.length - 1].split });

        selectedPreset: (document.querySelector(&#39;.preset-checkbox:checked&#39;)?.value || null)

function restoreAppState() {
    const stateStr = localStorage.getItem(&#39;paycal3_app_state&#39;);
    if (!stateStr) return;
    try {
        const state = JSON.parse(stateStr);
        // Format
        if (state.format === &#39;doubles&#39;) {
            document.getElementById(&#39;formatDoubles&#39;).checked = true;
            document.getElementById(&#39;formatSingles&#39;).checked = false;
        } else {
            document.getElementById(&#39;formatSingles&#39;).checked = true;
            document.getElementById(&#39;formatDoubles&#39;).checked = false;
        }
        // Set all UI fields before triggering any logic
        // Doubles type
        if (document.getElementById(&#39;doublesType&#39;)) {
            document.getElementById(&#39;doublesType&#39;).value = state.doublesType || &#39;same&#39;;
        }
        // Entry fee and players
        document.getElementById(&#39;entryFee&#39;).value = state.entryFee;
        document.getElementById(&#39;numPlayers&#39;).value = state.numPlayers;
        // Deductions
        document.getElementById(&#39;enableDartConnectFee&#39;).checked = !!state.enableDartConnectFee;
        document.getElementById(&#39;dartConnectFee&#39;).value = state.dartConnectFee;
        document.getElementById(&#39;enableHighScoreCont&#39;).checked = !!state.enableHighScoreCont;
        document.getElementById(&#39;highScoreCont&#39;).value = state.highScoreCont;
        document.getElementById(&#39;enableHighFinishCont&#39;).checked = !!state.enableHighFinishCont;

        /**
         * Rounds an amount to the nearest whole dollar ($1.00).
         * @param {number} amount - The amount to round.
         * @returns {number} The rounded amount (0 decimal places).

        function roundToNearestDollar(amount) {
            return Math.round(amount);

        // Function to determine the starting rank number for sorting (e.g., &#39;1st&#39;-&gt;1, &#39;3rd-4th&#39;-&gt;3, &#39;9th-16th&#39;-&gt;9)
        function getStartRank(place) {
            const match = place.match(/^(\d+)/);
            return match ? parseInt(match[1]) : Infinity;
        }

        // Custom sorting function: Ranks from 1st downwards
        function sortPrizes(a, b) {
            const rankA = getStartRank(a.place);
            const rankB = getStartRank(b.place);

            if (rankA &lt; rankB) return -1;
            if (rankA &gt; rankB) return 1;
            return 0; // Should not happen with well-defined ranks
        }
        
        // Helper function to convert long rank labels (e.g., &#39;3rd-4th&#39;) into short joint labels (e.g., &#39;Jnt 3rd&#39;)
        function getDisplayRankLabel(place, split) {
            if (split === 1) {
                return place; // e.g., &#39;1st&#39;, &#39;2nd&#39;, &#39;New Place&#39;
            }
            // For split ranks (e.g., &#39;3rd-4th&#39;, &#39;5th-8th&#39;), extract the starting rank and prepend &#39;Jnt&#39;
            const match = place.match(/^(\d+)(st|nd|rd|th)/);
            if (match) {
                // Extracts &#39;3rd&#39; from &#39;3rd-4th&#39;
                return `Jnt ${match[0]}`; // e.g., &#39;Jnt 3rd&#39;, &#39;Jnt 5th&#39;
            }
            return place; // Default for &#39;New Place&#39; or other custom labels
        }

        function applyPreset(presetName) {
            const entryFee = parseFloat(document.getElementById(&#39;entryFee&#39;).value) || 0;
            const numPlayers = parseInt(document.getElementById(&#39;numPlayers&#39;).value) || 0;

            if (presetName === &#39;top4_buyin_refund&#39;) {
                // Top 4 is a special case: 50/25/12.5/12.5 split
                prizes = PRESETS[presetName].map(p =&gt; ({ ...p }));
            } else if (presetName.endsWith(&#39;_buyin_refund&#39;)) {
                // Dynamic Money Back logic for Top 8, 16, 32
                const n = parseInt(presetName.match(/\d+/)[0]);
                const ranks = PRESETS.createRanks(n);

                // Determine if doubles is selected
                const isDoubles = document.getElementById(&#39;formatDoubles&#39;)?.checked;

                // For singles: refund is entryFee per player; for doubles: refund is entryFee/2 per player (team gets entryFee)
                const numRefundTeams = ranks[ranks.length - 1].split; // e.g., 4 for top 8, 8 for top 16
                let totalRefundAmount;
                if (isDoubles) {
                    totalRefundAmount = numRefundTeams * entryFee; // Each team gets entryFee (so each player gets entryFee/2)
                } else {
                    totalRefundAmount = numRefundTeams * entryFee; // Each player gets entryFee
                }

                // Get the net prize pool for scaling, which respects the current deduction setting
                const dartConnectEnabled = document.getElementById(&#39;enableDartConnectFee&#39;).checked;
                const highScoreEnabled = document.getElementById(&#39;enableHighScoreCont&#39;).checked;
                const highFinishEnabled = document.getElementById(&#39;enableHighFinishCont&#39;).checked;

                const dartConnectInput = parseFloat(document.getElementById(&#39;dartConnectFee&#39;).value) || 0;
                const highScoreInput = parseFloat(document.getElementById(&#39;highScoreCont&#39;).value) || 0;
                const highFinishInput = parseFloat(document.getElementById(&#39;highFinishCont&#39;).value) || 0;

                const dartConnectFee = dartConnectEnabled ? dartConnectInput : 0;
                const highScoreCont = highScoreEnabled ? highScoreInput : 0;
                const highFinishCont = highFinishEnabled ? highFinishInput : 0;

                const totalDeductionPerPlayer = dartConnectFee + highScoreCont + highFinishCont;
                const netEntryFee = Math.max(0, entryFee - totalDeductionPerPlayer);
                const numPlayers = parseInt(document.getElementById(&#39;numPlayers&#39;).value) || 0;
                const netMainPrizePool = netEntryFee * numPlayers;

                if (netMainPrizePool === 0) {
                    prizes = []; // Cannot calculate if pool is 0
                    calculatePrizes();
                    return;
                }

                // Calculate the percentage needed for the refund tier (based on NET pool)
                const lastPercent = (totalRefundAmount / netMainPrizePool) * 100;

                // Determine the percentage remaining for the top places
                const remainingPercent = 100 - lastPercent;

                // Define standard ratios for the top ranks after reserving the refund amount (30% Winner base)
                const ratios = {
                    &#39;top8&#39;: [30.00, 20.00, 20.00], // Ratios for 1st, 2nd, 3rd-4th
                    &#39;top16&#39;: [30.00, 15.00, 15.00, 20.00], // Ratios for 1st to 5th-8th
                    &#39;top32&#39;: [30.00, 15.00, 10.00, 15.00, 15.00], // Ratios for 1st to 9th-16th
                };
                const ratioKey = presetName.substring(0, presetName.indexOf(&#39;_&#39;));
                const baseRatios = ratios[ratioKey] || [];
                const baseRatioTotal = baseRatios.reduce((sum, ratio) =&gt; sum + ratio, 0);

                prizes = [];

                if (baseRatioTotal &gt; 0 &amp;&amp; remainingPercent &gt; 0) {
                    // 4. Scale up the top rank percentages using the remaining pool
                    for (let i = 0; i &lt; baseRatios.length; i++) {
                        const scaledPercent = (baseRatios[i] / baseRatioTotal) * remainingPercent;
                        prizes.push({ place: ranks[i].place, percent: scaledPercent, split: ranks[i].split });
                    }
                } else if (remainingPercent &gt; 0 &amp;&amp; baseRatioTotal === 0) {
                    // Fallback for cases where baseRatios is empty but there&#39;s a remaining pool
                    prizes.push({ place: &#39;1st&#39;, percent: remainingPercent, split: 1 });
                }

                // 5. Add the final refund tier (if lastPercent is positive)
                if (lastPercent &gt; 0) {
                    prizes.push({ place: ranks[ranks.length - 1].place, percent: lastPercent, split: ranks[ranks.length - 1].split });
                }

            } else {
                // Standard static percentage presets (50% and 30% Winner)
                prizes = PRESETS[presetName].map(p =&gt; ({ ...p }));
            }

            // Only update UI when Calculate is pressed
        }
